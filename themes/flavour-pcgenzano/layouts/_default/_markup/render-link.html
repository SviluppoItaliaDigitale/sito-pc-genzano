{{- $link := .Destination -}} {{- if strings.HasPrefix $link "/" -}} {{- /* Link interno: estraggo il path senza fragment/query per la lookup */ -}} {{- $path := $link -}} {{- if in $path "#" }}{{ $path = index (split $path "#") 0 }}{{ end -}} {{- if in $path "?" }}{{ $path = index (split $path "?") 0 }}{{ end -}} {{- /* Se il link punta a un file statico (estensione nota), lo rendo direttamente senza lookup di pagina */ -}} {{- $staticExts := slice ".pdf" ".webp" ".jpg" ".jpeg" ".png" ".gif" ".svg" ".doc" ".docx" ".xls" ".xlsx" ".ppt" ".pptx" ".zip" ".mp3" ".mp4" ".csv" ".txt" ".rtf" ".html" ".htm" -}} {{- $lowerPath := lower $path -}} {{- $isStatic := false -}} {{- range $staticExts -}} {{- if strings.HasSuffix $lowerPath . -}}{{- $isStatic = true -}}{{- end -}} {{- end -}} {{- /* Cartella statica con index.html (es. /giochi/infanzia/, /abili-a-proteggere/) */ -}} {{- if and (not $isStatic) (strings.HasSuffix $path "/") -}} {{- $staticIdx := printf "static%sindex.html" $path -}} {{- if fileExists $staticIdx -}}{{- $isStatic = true -}}{{- end -}} {{- end -}} {{- /* Hugo relURL non aggiunge il subpath del baseURL ai path con leading /, quindi togliamo lo slash iniziale: "cerca/" → "/sito-pc-genzano/cerca/" mentre "/cerca/" → "/cerca/" (rotto su GitHub Pages). */ -}} {{- $relLink := strings.TrimPrefix "/" $link -}} {{- if $isStatic -}} {{ .Text | safeHTML }} {{- else -}} {{- $cleanPath := strings.TrimSuffix "/" (strings.TrimPrefix "/" $path) -}} {{- $target := site.GetPage $cleanPath -}} {{- if $target -}} {{ .Text | safeHTML }} {{- else -}} {{ .Text | safeHTML }} {{- end -}} {{- end -}} {{- else if strings.HasPrefix $link "http" -}} {{ .Text | safeHTML }} {{- else if or (strings.HasPrefix $link "tel:") (strings.HasPrefix $link "mailto:") -}} {{ .Text | safeHTML }} {{- else -}} {{ .Text | safeHTML }} {{- end -}}
